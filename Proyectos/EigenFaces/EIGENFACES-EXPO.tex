\subsection{Algoritmo de PCA}
\begin{frame}
	 \begin{proof}[Modelo]
	 	\begin{enumerate}
	 		\item {\bf INPUT:} $\phi_1,\ldots,\phi_K$ (K datos $n$ dimensionales).
	 		\item Normalización de la media. $\Phi_i=\phi_i - \overline{\phi}$
	 		\item SVD aplicado a los datos normalizados
	 		\begin{itemize}
	 			\item Eigenvalores de la matriz de covarianza de los datos (Varianza explicada)
	 			\item Eigenvectores de la matriz de covarianza de los datos
	 		\end{itemize}
	 		\item Determinación del número de componentes principales
	 		\item Determinación de los componentes principales $u_1,\ldots,u_m$ (eigenvectores que mantienen mayor varianza de los datos).
	 		\item Patrones de clasificación: $\begin{bmatrix}
	 		\Theta_1 & \ldots &\Theta_K
	 		\end{bmatrix} = \begin{bmatrix}
	 		u_1^T\\
	 		\vdots\\
	 		u_m^T
	 		\end{bmatrix}\begin{bmatrix}
	 		\Phi_1 & \ldots & \Phi_K
	 		\end{bmatrix}$.
	 	\end{enumerate}
	 \end{proof}
\end{frame}

\subsection{Eigenfaces}
\begin{frame}
	\begin{defiei}
		Se denominan eigenfaces a los eigenvectores, de la matriz de covarianza de las imagenes faciales $\phi_1,\ldots,\phi_K$, que generan el subespacio de dimensión reducida usando el análisis de componentes principales.
	\end{defiei}
	\begin{figure}[h]
		\centering
		\includegraphics[width=100mm]{eigenf}
		%\caption{\small Maching L}
	\end{figure}
\end{frame}
\begin{frame}
	\begin{defifs}
			Se denomina face-space al subespacio generado por los eigenfaces.
	\end{defifs}
\end{frame}

\begin{frame}{Matriz de Covarianza.}

 $$ C=\dfrac{1}{K}\begin{bmatrix}
  \Phi_1 & \Phi_2 & \ldots & \Phi_K
  \end{bmatrix}_{(n\times m)\times K}\begin{bmatrix}
  \Phi_1^T\\
  \Phi_2^T\\
  \vdots\\
  \Phi_K
  \end{bmatrix}_{K\times (n\times m)}
  $$
  cuya dimensión es $(n\times m)^2$.
\end{frame}
\begin{frame}{Matriz de Covarianza.}
	\begin{teo}
		Si $v$ es un eigenvector, con valor propio $\lambda\neq 0$, de $A^{T}A$, luego $Av$ es un eigenvector con el mismo valor propio $\lambda$ de $AA^{T}$.
	\end{teo}
	$$
	D=\dfrac{1}{K}\begin{bmatrix}
	\Phi_1^T\\
	\Phi_2^T\\
	\vdots\\
	\Phi_K
	\end{bmatrix}_{K\times (n\times m)}\!\!\!\!\!\!\!\!\!\!\!\!\!\!\begin{bmatrix}
	\Phi_1 & \Phi_2 & \ldots & \Phi_K
	\end{bmatrix}_{(n\times m)\times K}
	$$
	Por tanto siendo $v_1,v_2,\ldots,v_{K'}$ $(K'\leq K)$ eigenvectores de $D$, asociados a eigenvalores $\lambda_1,\ldots,\lambda_{K'}$  distintos de cero, los eigenvectores de $C$ asociados a los mismo autovalores distintos de cero son $\{Av_1,Av_2,\ldots,Av_{K'}\}$ siendo $A=[\Phi_1,\ldots,\Phi_K]$. Denotemos $u_i=Avi$\\
\end{frame}
\begin{frame}{Pérdida de Variabilidad}
	\begin{eqnarray}
	\sum_{i=1}^{K'}\lambda_i & \mbox{ (Varianza total en $R^{K'}$)}\\
	\sum_{i=1}^{n\times m}\lambda_i = \sum_{i=1}^{K'}\lambda_i& \mbox{ (Varianza total en $R^{n\times m}$)}
	\end{eqnarray}
	Por tanto el porcentaje de varianza que se mantiene en el subespacio de dimensión $R^{K'}$ es
	\begin{eqnarray}
	\dfrac{\mbox{Varianza total en $R^{K}$}}{\mbox{Varianza total en $R^{n\times m}$}}\times 100\% & =& 100\%.
	\end{eqnarray}
 
\end{frame}
\begin{frame}{¿Podríamos entonces decir que los eigenvectores $\{u_1,u_2,\ldots,u_{K'}\}$ son los eigenfaces?}En la práctica realizada por Sirivich y Kirby (1987), se observó que tomando los eigenvectores $u_1,u_2,\ldots,u_{115}$ de la matriz de covarianza de sus datos que mantengan el 100\%  de variabilidad, solo los primeros $40$ eigenvectores fueron suficientes para una excelente representación pues se generaba una pérdida del 2\% de variabilidad. Por este hecho, es útil el PCA para determinar el número de componentes principales que no genere pérdida de variabilidad significante, digamos no más del $2 \%$
	
\end{frame}

\section{Reconocimiento Facial}
\subsection{Clasificación de las Imágenes Faciales.}
\begin{frame}
	Sea  $\phi$ imagen de entrada, ésta puede ser clasificada como
  \begin{enumerate}
  	\item {\bf Reconocimiento de una imagen facial}:
  	\item {\bf No reconocimiento de  una imagen facial}:
  	\item {\bf Imagen no facial}
  \end{enumerate}
  De la imagen de entrada.\\
  {\bf Normalizando:}  $\Phi=\phi-\Psi$.
  {\bf Dato transformado:}
  $
  \Theta = \begin{bmatrix}
  u_1^T\Phi\\
  u_2^T\Phi\\
  \vdots\\
  u_{K'}^T\Phi
  \end{bmatrix}
  $
  De las $K$ imágenes faciales.\\
  {\bf Patrones de clasificación:}$\begin{bmatrix}
  \Theta_1 & \ldots &\Theta_K
  \end{bmatrix} = \begin{bmatrix}
  u_1^T\\
  \vdots\\
  u_m^T
  \end{bmatrix}\begin{bmatrix}
  \Phi_1 & \ldots & \Phi_K
  \end{bmatrix}$
\end{frame}
\begin{frame}{Imagen facial- No facial}
	Nuestra verificación inicial es determinar si la imagen de entrada es facial o no lo es, esto es si el face-space la clasifica como tal
	$$
	\varepsilon^2=\|\Phi-proy_{\;F}\Phi\|^2\leq \delta
	$$
	donde  $proy_{F}(\Phi)= \sum_{i=1}^{K'}(u_i\cdot\Phi)u_i$.
	\begin{figure}[h]
		\centering
		\subfigure[El subespacio clasifica el dato.]{\includegraphics[width=30mm]{cl2}}
		\subfigure[El subespacio no clasifica el dato]{\includegraphics[width=30mm]{cl1}}
		\caption{\small Ejemplo de un subespacio bidimensional, caracterizando un dato tridimensional.}
	\end{figure}
\end{frame}
\begin{frame}{Reconocimiento Facial}
	determinaremos el patron ${\Theta_i}$, que mejor clasifique a ${\Theta}$, usando el mismo análisis anterior:

	$$\epsilon^2 = \min\{\epsilon^2_i:\;\|\Theta-\Theta_i\|^2=\epsilon^2_i, i=1,\ldots,K' \}.$$
	Dado este hecho, bajo medios empíricos, también se recomienda establecer un límite $\delta$ a esta cantidad.
	
	$$\epsilon^2\leq \delta$$
	Por tanto si $\epsilon^2\leq\delta$, el patrón $\Phi_i$ estableciéndose lo que denominamos {\bf reconocimiento facial}, y por tanto la imagen facial $\phi$ matendría una alta relación con la imagen $\phi_i$, debido a que se pierde muy poca variabilidad entre ellas. \\
	En caso contrario no se establece un reconocimiento facial.
\end{frame}	
\begin{frame}{Ajuste del límite para la clasificación}
	\begin{figure}[h]
		\centering
		\includegraphics[width=70mm]{LIM1}
		\caption{\small X: $\delta$;Y: Número de clasificaciones correctas. Tomándose 16 imágenes faciales, de las $K$ imágenes faciales de entradas,  pero variando su: ($a$) iluminación - ($b$) inclinación de su rostro - ($c$) tamaño de cabeza - ($d$) iluminación e inclinación - ($e$) iluminación y tamaño de cabeza - ($f$) inclinación y tamaño de cabeza.}
	\end{figure}
\end{frame}
\subsection{Modelo del Algoritmo}
\begin{frame}{Entrenamiento}
	\begin{itemize}
			\item Muestra de $K$ de imágenes faciales, como datos de entrenamiento para el aprendizaje
			\item Aplicar PCA para obtener los eigenvectores de la matriz $D$ determinan no más del $2\%$ de pérdida de variabilidad. Estos eigenvectores serán llamados {\it eigenfaces}.
			\item Determinar los patrones de clasificación transformados.
		\end{itemize}
\end{frame}
\begin{frame}{Tarea de reconocimiento}
		\begin{itemize}
			\item Imagen de entrada
			\item Dado un límite $\delta$, verificar si se cumple:
			$$
			\varepsilon^2=\|\Phi-proy_{\;F}\Phi\|^2\leq \delta
			$$
			donde  $proy_{F}(\Phi)= \sum_{i=1}^{K'}(u_i\cdot\Phi)u_i$.
			\item De cumplirse lo anterior se clasifica como una imagen facial, en caso contrario se clasifica como imagen no facial.
			\item Dado $\delta>0$, verificar si se cumple:
			$$\epsilon^2 = \min\{\epsilon^2_i:\;\|\Theta-\Theta_i\|^2=\epsilon^2_i, i=1,\ldots,K' \}\leq \delta.$$
			\item De cumplirse lo anterior se determina un reconocimiento facial y el patrón $\Theta_i$ que la caracterize, indicará la alta relación entre $\phi$ con la imagen facial $\phi_i$, a causa de la poca pérdida de variabilidad. De no cumplirse no existe un reconocimiento facial.
		\end{itemize}
\end{frame}
\begin{frame}{Tareas adicionales}
		\begin{itemize}
			\item De clasificar la imagen de entrada como imagen facial, estableciéndose o no el reconocimiento facial, entonces implementarla en el modelo actualizando el conjunto de las $K$ imágenes faciales de entrenamiento.
			\item  De clasificar la imagen de entrada como imagen facial, no estableciéndose el reconocimiento facial, entonces adquirir su dato transformado e implementar en la lista de patron de clasificación.
		\end{itemize}
\end{frame}
